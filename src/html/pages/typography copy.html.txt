        <script>
            const fontSizes = {
                15: "clamp(1.3rem, 0.9vw, 1.5rem)",
                16: "clamp(1.4rem, 0.9vw, 1.6rem)",
                18: "clamp(1.5rem, 1vw, 1.8rem)",
                20: "clamp(1.5rem, 1.1vw, 2rem)",
                24: "clamp(1.6rem, 1.3vw, 2.4rem)",
                26: "clamp(1.6rem, 1.4vw, 2.6rem)",
                30: "clamp(1.8rem, 1.6vw, 3rem)",
                32: "clamp(1.8rem, 1.8vw, 3.2rem)",
                44: "clamp(2rem, 2.4vw, 4.4rem)",
                50: "clamp(2rem, 2.7vw, 5rem)",
                60: "clamp(2rem, 3.2vw, 6rem)",
                70: "clamp(2.5rem, 3.8vw, 7rem)",
                120: "clamp(4rem, 6.5vw, 12rem)"
            };

            const tableBody2 = document.getElementById("font-table-body");
            const remToPx2 = (rem) => rem * 10; // 1rem = 10px

            for (const [key, clampValue] of Object.entries(fontSizes)) {
                const match = clampValue.match(/clamp\(([^,]+),\s*([^,]+),\s*([^)]+)\)/);
                if (!match) continue;

                let [_, minRaw, vwRaw, maxRaw] = match;

                const parseRem = (val) => parseFloat(val.replace("rem", ""));
                const parseVw = (val) => parseFloat(val.replace("vw", ""));

                const minRem = parseRem(minRaw);
                const maxRem = parseRem(maxRaw);
                const vw = parseVw(vwRaw);

                const minPx = remToPx2(minRem);
                const maxPx = remToPx2(maxRem);

                const minScreen = Math.round((minPx / vw) * 100);
                const maxScreen = Math.round((maxPx / vw) * 100);

                const row = document.createElement("tr");
                row.innerHTML = `
            <td><code>.cdx-text-${key}</code> <span class="cdx-color-dark cdx-text-${key}">Text ${key}</span></td>
            <td>${minPx}px = ${minRem}rem</td>
            <td>${vw}vw</td>
            <td>${maxPx}px = ${maxRem}rem</td>
            <td>${minScreen}px</td>
            <td>${maxScreen}px</td>
        `;
                tableBody2.appendChild(row);
            }
        </script>

            const remToPx = (rem) => rem * 10; // 1rem = 10px (custom scale)
